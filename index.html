function signUp(username, password, adminUser, adminPass){
  var signedInInfo = signInAsAdmin(adminUser, adminPass);
  var adminToken = signedInInfo.map('.token');
  var newUserStatus = createUser(adminToken, username, password);
  return newUserStatus;
}

function createUser(adminToken, username, password){
  var newUserData = adminToken.flatMap(createNewUser, username, password);
  var newUserKey = newUserData.map('.key');
  var newUserAuthKey = 
		Bacon.combineWith(addAuthToUser, adminToken, 
                      username, password, 
                      newUserKey).flatMapLatest(sendRequest);
  return newUserAuthKey;
}

Experience of Using FSIO

Overall: Quite Positive

  • Mapping grocery items <--> files worked well
  • Low latency
  • Clear documentation
  • Easy to use

Minor Issues

  • No journal tracking of metadata (--> notifications before scanning complete)
  • Various minor documentation issues
  • Firefox error messages on certain FSIO responses

Deployment-related Issues

  • Confusion around URLs
  • No way to manage users

Future Development Ideas

  • Pairing with other users (Collaborations API)
  • Add items from own recipes
  • Sort list by categories e.g. store section
  • Share anonymously with another user (Shares/proxying)
  • Recipe images
